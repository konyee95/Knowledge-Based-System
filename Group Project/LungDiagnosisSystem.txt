(defrule tuberculosis
	(and
	(persistent_cough yes)
	(constant_fatigue yes)
	(weight_loss yes)
	(lack_of_apetite yes)
	(fever yes)
	(coughing_blood yes)
	(night_sweats yes))
	=> (printout t "Diagnosing a lung disease..............." crlf)
	(printout t "That lung disease could be tuberculosis." crlf)
)
